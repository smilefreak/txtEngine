00351         // Get a character, while interpreting entities.
00352         // The length can be from 0 to 4 bytes.
00353         inline static const char* GetChar( const char* p, char* _value, int* length, TiXmlEncoding encoding ) {
00354             assert( p );
00355             if ( encoding == TIXML_ENCODING_UTF8 ) {
00356                 *length = utf8ByteTable[ *((const unsigned char*)p) ];
00357                 assert( *length >= 0 && *length < 5 );
00358                 }
00359             else {
00360                 *length = 1;
00361                 }
00362 
00363             if ( *length == 1 ) {
00364                 if ( *p == '&' )
00365                     return GetEntity( p, _value, length, encoding );
00366                 *_value = *p;
00367                 return p+1;
00368                 }
00369             else if ( *length ) {
00370                 //strncpy( _value, p, *length );        // lots of compilers don't like this function (unsafe),
00371                 // and the null terminator isn't needed
00372                 for( int i=0; p[i] && i<*length; ++i ) {
00373                     _value[i] = p[i];
00374                     }
00375                 return p + (*length);
00376                 }
00377             else {
00378                 // Not valid text.
00379                 return 0;
00380                 }
00381             }
